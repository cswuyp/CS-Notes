# 1. 简单多线程实例
```c++
#include <iostream>
#include<thread>
using namespace std;
void foo()
{
	for (int i = 0; i < 100; i++)
	{
		cout << "foo() i:" << i << endl;
	}
}
void bar(int x)
{
	for (int j = x; j < 200; j++)
	{
		cout << "bar():j" << j << endl;
	}
}
int main()
{
	thread first(foo);
	thread second(bar,100);
	first.join();
	second.join();
	return 0;
}
```
